
SET(TIMEZONECONFIGURATION_MAJOR "0")
SET(TIMEZONECONFIGURATION_MINOR "0")
SET(TIMEZONECONFIGURATION_PATCH "1")
SET(TIMEZONECONFIGURATION_VERSION "${TIMEZONECONFIGURATION_MAJOR}.${TIMEZONECONFIGURATION_MINOR}.${TIMEZONECONFIGURATION_PATCH}")

IF(APPLE)
	SET(CMAKE_INSTALL_PREFIX $ENV{HOME}/Library/Application\ Support/Stellarium)
ElSE(APPLE)
	SET(CMAKE_INSTALL_PREFIX $ENV{HOME}/.stellarium)
ENDIF(APPLE)

ADD_DEFINITIONS(-DPLUGIN_VERSION="${TIMEZONECONFIGURATION_VERSION}")

ADD_SUBDIRECTORY( src )

INSTALL(FILES DESTINATION "modules/TimeZoneConfiguration")
